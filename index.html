                bool esSumaConjunto(int conjunto[], int n, int sum){
                
                    //* Casos base
                    if( sum == 0 )
                        return true;
                    if( sum < 0 || (n == 0 && sum != 0) )
                        return false;
                
                    //* Si el último elemento es mayor a la suma que estamos verificando se ignora
                    if( conjunto[n-1] > sum )
                        return esSumaConjunto(conjunto, n-1, sum);

                }
